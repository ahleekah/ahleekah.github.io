// gamemanager.cpp
#include "GameManager.hpp"
#include <iostream>
#include "player.hpp"

namespace SDK {

    GameManager::GameManager() : 
        player("Player1", 100.0, "Pistol"),
        window(sf::VideoMode(800,600), "SFML Game") {
        sf::Vector2u windowSize = window.getSize();

        player.getCharacterSprite().setOrigin(
            player.getCharacterSprite().getLocalBounds().width / 2, 
            player.getCharacterSprite().getLocalBounds().height / 2);
        player.getCharacterSprite().setPosition(windowSize.x / 2, windowSize.y / 2);
    }

    void GameManager::run() {
        while (window.isOpen()) {
            sf::Event event;
            while (window.pollEvent(event)) {
                if (event.type == sf::Event::Closed)
                    window.close();
            }

            sf::Vector2i mousePosition = sf::Mouse::getPosition(window);
            sf::Vector2f characterPosition = player.getCharacterSprite().getPosition();
            float angle = std::atan2(mousePosition.y - characterPosition.y, mousePosition.x - characterPosition.x);
            angle = angle * (180.0 / 3.14159265);

            player.getCharacterSprite().setRotation(angle - 90.0);
            window.clear(sf::Color::Black);
            window.draw(player.getCharacterSprite());
            window.display();
        }

    }
}
// main.cpp
#include "SDK/GameManager.hpp"

int main() {

    SDK::GameManager gameManager;
    gameManager.run();

    return 0;
}
